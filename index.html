// НАСТРОЙКИ const SPREADSHEET_ID = 'ваш_ид_таблицы'; // ВСТАВЬТЕ ID ВАШЕЙ ТАБЛИЦЫ const SHEET_NAME = 'Data'; const SETTINGS_SHEET = 'Settings'; // ОБРАБОТКА GET-ЗАПРОСОВ (получение текущего состояния) function doGet(e) { try { const state = getState(); return ContentService.createTextOutput(JSON.stringify(state)) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*") .addHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS") .addHeader("Access-Control-Allow-Headers", "Content-Type"); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ error: err.message })) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } } // ОБРАБОТКА POST-ЗАПРОСОВ (приём новой ставки) function doPost(e) { try { // Обработка preflight (OPTIONS) запроса if (e.method === 'OPTIONS') { return ContentService.createTextOutput('') .setMimeType(ContentService.MimeType.TEXT) .addHeader("Access-Control-Allow-Origin", "*") .addHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS") .addHeader("Access-Control-Allow-Headers", "Content-Type"); } // Парсим данные из запроса const data = JSON.parse(e.postData.contents); const userId = data.userId; const outcome = data.outcome; // "A" или "B" const betAmount = parseFloat(data.betAmount); // Валидация if (!["A","B"].includes(outcome)) { throw new Error("Неверный исход. Используйте A или B"); } if (isNaN(betAmount) || betAmount <= 0) { throw new Error("Сумма ставки должна быть положительным числом"); } // Получаем текущее состояние (чтобы узнать коэффициент для этой ставки) const stateBefore = getState(); let coeffAtBet; if (outcome === "A") { coeffAtBet = stateBefore.coeffA; } else { coeffAtBet = stateBefore.coeffB; } // Записываем ставку в таблицу const sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME); sheet.appendRow([ new Date(), // timestamp userId, // user_id outcome, // outcome betAmount, // bet_amount coeffAtBet // coefficient_at_bet (сохраняем кэф момента ставки) ]); // Получаем обновлённое состояние после записи const stateAfter = getState(); // Возвращаем успешный ответ const response = { success: true, message: `Ставка ${betAmount}₽ на исход ${outcome} принята. Коэффициент: ${coeffAtBet}`, state: stateAfter, yourBet: { outcome: outcome, amount: betAmount, coefficient: coeffAtBet, potentialWin: betAmount * coeffAtBet } }; return ContentService.createTextOutput(JSON.stringify(response)) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*") .addHeader("Access-Control-Allow-Methods", "GET, POST, OPTIONS") .addHeader("Access-Control-Allow-Headers", "Content-Type"); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON) .addHeader("Access-Control-Allow-Origin", "*"); } } // ПОЛУЧЕНИЕ ТЕКУЩЕГО СОСТОЯНИЯ (суммы ставок и коэффициенты) function getState() { // Открываем таблицу и лист с данными const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); // Считаем суммы ставок на исходы А и Б let sumA = 0; let sumB = 0; // Пропускаем заголовок (первая строка) for (let i = 1; i < data.length; i++) { const outcome = data[i][2]; // колонка C (индекс 2) const amount = parseFloat(data[i][3]); // колонка D (индекс 3) if (outcome === "A") { sumA += amount; } else if (outcome === "B") { sumB += amount; } } const total = sumA + sumB; // Читаем маржу из листа настроек (по умолчанию 10% = 0.1) let margin = 0.1; try { const settingsSheet = ss.getSheetByName(SETTINGS_SHEET); if (settingsSheet) { const marginCell = settingsSheet.getRange("B1").getValue(); if (!isNaN(parseFloat(marginCell)) && marginCell !== "") { margin = parseFloat(marginCell); } } } catch(e) { // Если листа настроек нет, используем маржу 0.1 } // Расчёт справедливых коэффициентов (без маржи) let fairCoeffA = (total === 0 || sumA === 0) ? 1.0 : total / sumA; let fairCoeffB = (total === 0 || sumB === 0) ? 1.0 : total / sumB; // Применяем маржу let coeffA = fairCoeffA * (1 - margin); let coeffB = fairCoeffB * (1 - margin); // Защита: коэффициент не может быть ниже 1.01 и выше 100 coeffA = Math.max(1.01, Math.min(100.0, coeffA)); coeffB = Math.max(1.01, Math.min(100.0, coeffB)); // Округление до 2 знаков после запятой coeffA = Math.round(coeffA * 100) / 100; coeffB = Math.round(coeffB * 100) / 100; // Возвращаем объект с состоянием return { total: total, poolA: sumA, poolB: sumB, coeffA: coeffA, coeffB: coeffB, margin: margin * 100, // возвращаем в процентах для отображения timestamp: new Date().toISOString() }; } // Функция для тестирования (можно запустить в редакторе) function test() { console.log(getState()); }
